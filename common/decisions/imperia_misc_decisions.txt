become_pretender_israel = {
	is_shown = {
		country_has_primary_culture = cu:beta_israel
		NOT = {
			has_variable = is_pretender_israel
		}
		NOT = {
			exists = c:ISR
		}
		any_country = {
			has_variable = is_pretender_israel
			count = 0
		}
	}
	possible = {
		is_at_war = no
		has_technology_researched = nationalism
		country_has_state_religion = rel:jewish
	}
	when_taken = {
		set_variable = is_pretender_israel
		add_primary_culture = cu:ashkenazi
		add_primary_culture = cu:sephardic
	}
	ai_chance = {
		value = 0
	}
}
