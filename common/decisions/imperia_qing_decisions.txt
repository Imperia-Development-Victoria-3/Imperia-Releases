qing_language_call_language_conference = {
	is_shown = {
		exists = c:CHI
		c:CHI = ROOT
		is_subject = no
		not = {
			has_variable = has_called_chinese_language_conference
		}
	}
	possible = {
		c:CHI = {
			has_technology_researched = nationalism
			custom_tooltip = {
				text = chinese_national_identity_tt
				has_variable = je_chinese_national_identity_completed
			}
		}
	}
	when_taken = {
		trigger_event = {
			id = imperia_qing_substates.6
			days = 0
			popup = yes
		}
	}
	ai_chance = {
		value = 15
	}
}

warlord_china_declare_rival_gov = {
	is_shown = {
		NOT = {
			exists = c:CHI
		}
		has_global_variable = china_shatters
		NOR = {
			has_modifier = modifier_recognized_national_government
			has_modifier = modifier_rival_national_government
		}
		OR = {
			capital = {
				state_is_in_china = yes
			}
			country_has_primary_culture = cu:manchu
		}
	}
	possible = {
		NOT = {
			exists = c:CHI
		}
		any_scope_state = {
			state_is_in_china = yes
			count >= 5
		}
		any_scope_state = {
			OR = {
				state_region = s:STATE_BEIJING
				state_region = s:STATE_XIAN
				state_region = s:STATE_NANJING
				state_region = s:STATE_GUANGDONG
			}
		}
	}
	when_taken = {
		add_modifier = {
			name = modifier_rival_national_government
		}
		add_journal_entry = {
			type = je_warlord_overthrow_national_gov
		}
	}
}

qing_invite_boxer_leaders_to_forbidden_city = {
	is_shown = {
		c:CHI ?= root
		has_technology_researched = pan-nationalism
		country_has_primary_culture = cu:manchu
		NOT = {
			has_variable = invited_boxer_leaders_to_capital
		}
	}
	possible = {
		any_country = {
			has_technology_researched = political_agitation
		}
		custom_tooltip = {
			text = je_hundred_days_reform_failed_tt
			has_variable = je_hundred_days_reform_failed
		}
	}
	when_taken = {
		set_variable = invited_boxer_leaders_to_capital	# This will be removed if you press no on the confirmation event, allowing you to press the button again
		custom_tooltip = boxer_rebellion_warning_tt
		trigger_event = {	# Invite Boxer Leaders confirmation event
			id = imperia_qing.8
			popup = yes
		}
	}
	ai_chance = {
		value = 15
	}
}
