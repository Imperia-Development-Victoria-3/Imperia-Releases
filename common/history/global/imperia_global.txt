GLOBAL = {
	# Decentralised countries are no longer included in every_country apparently, as of 1.9 or so??
	every_decentralized_country = {
		## ADD TRIBAL LEVIES TO ELIGIBLE COUNTRIES
		if = {
			limit = {
				can_have_tribal_levies = yes
			}
			activate_law = law_type:law_tribal_levies
		}
	}
	every_country = {
		### Infamy setup ###
		set_variable = {
			name = infamy_on_month_start
			value = infamy
		}
		### GAME START BUREAUCURACY BONUS ###
		add_modifier = {
			name = game_start_bureaucracy_bonus
			is_decaying = yes
			months = 121
		}
		### MONEY TRANSFER ###
		set_variable = {
			name = imperia_money_transfer_amount
			value = 0
		}
		### SET STARTING FOREIGN POLICY ###
		update_foreign_policy = yes
		remove_variable ?= modifier_imperia_foreign_policy_cooldown	# To allow players to change from the get-go
		### TECHNOLOGIES ###
		if = {
			limit = {
				any_state = {
					is_coastal = yes
				}
			}
			add_technology_researched = age_of_sail
		}
		if = {
			limit = {
				has_technology_researched = screw_frigate
			}
			add_technology_researched = propellers
		}
		## ADD TRIBAL LEVIES TO ELIGIBLE COUNTRIES
		if = {
			limit = {
				can_have_tribal_levies = yes
			}
			activate_law = law_type:law_tribal_levies
		}
		### JEWISH ETHIOPIA ###
		if = {
			limit = {
				any_scope_pop = {
					religion = rel:jewish
					culture = {
						has_discrimination_trait = heritage_abyssinian
					}
				}
			}
			every_scope_pop = {
				limit = {
					religion = rel:jewish
					culture = {
						has_discrimination_trait = heritage_abyssinian
					}
				}
				# Done like this so we dont have to import the african pop history file...
				change_pop_culture = {
					target = cu:beta_israel
					value = 1
				}
			}
		}
	}
	### STRATEGIC REGIONS LIST FOR VIENNA CONGRESS ###
	make_strategic_region_list = yes
	# British East India Company subjects start with a mortality modifier
	c:BIC ?= {
		every_subject_or_below = {
			limit = {
				capital = {
					OR = {
						region = sr:region_madras
						region = sr:region_bombay
						region = sr:region_bengal
						region = sr:region_central_india
						region = sr:region_punjab
						region = sr:region_himalayas
						state_region = s:STATE_MANDALAY
						state_region = s:STATE_KACHIN
						state_region = s:STATE_SHAN_STATES
						state_region = s:STATE_PEGU
					}
				}
			}
			add_modifier = {
				name = imperia_british_india_mortality_modifier
				is_decaying = yes
				years = 30
			}
		}
		add_modifier = {
			name = imperia_british_india_mortality_modifier
			is_decaying = yes
			years = 30
		}
		### Foreign Policy GUI bandaid ###
		make_list_of_foreign_policies = yes
	}
}
