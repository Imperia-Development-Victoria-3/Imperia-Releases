### VTM ###
vtm_je_italian_unification_button = {
	name = "vtm_je_italian_unification_savoy"
	desc = "vtm_je_italian_unification_savoy_desc"
	visible = {
		exists = c:FRA
		NOT = {
			has_global_variable = fra_annexed_savoy
		}
		
		any_scope_state = {
			state_region = s:STATE_SAVOY
			is_split_state = no
			is_capital = no
		}
		is_subject = no
	}
	
	possible = {
		has_technology_researched = nationalism
		c:FRA ?= {
			is_at_war = no
			owns_entire_state_region = STATE_RHONE
			is_subject = no
			NOT = {
				owes_obligation_to = ROOT
			}
			relations:root >= relations_threshold:cordial
		}
	}
	
	effect = {
		set_global_variable = fra_annexed_savoy
		set_global_variable = fra_savoy_sardinia
		c:FRA ?= {
			set_owes_obligation_to = {
				country = ROOT
				setting = yes
			}
		}
		
		show_as_tooltip = {
			random_scope_state = {
				limit = {
					state_region = s:STATE_SAVOY
				}
				set_state_owner = c:FRA
			}
			
			if = {
				limit = {
					any_scope_state = {
						state_region = s:STATE_PROVENCE
					}
				}
				
				random_scope_state = {
					limit = {
						state_region = s:STATE_PROVENCE
					}
					set_state_owner = c:FRA
				}
			}
		}
		
		c:FRA ?= {
			trigger_event = {
				id = natural_borders.3
				popup = yes
			}
		}
		
		hidden_effect = {
			c:FRA ?= {
				set_secret_goal = {
					country = ROOT
					secret_goal = befriend
				}
			}
		}
	}
	
	ai_chance = {
		base = 0
		modifier = {
			trigger = {
				OR = {
					has_diplomatic_pact = {
						who = c:FRA
						type = defensive_pact
					}
					
					has_diplomatic_pact = {
						who = c:FRA
						type = alliance
					}
				}
			}
			add = 2
		}
		
		modifier = {
			trigger = {
				c:FRA ?= {
					relations:root >= relations_threshold:cordial
				}
			}
			add = 1
		}
		
		modifier = {
			trigger = {
				c:FRA ?= {
					relations:root >= relations_threshold:amicable
				}
			}
			add = 1
		}
		
		modifier = {
			trigger = {
				country_rank >= rank_value:major_power
			}
			add = 1
		}
		
		modifier = {
			trigger = {
				NOT = {
					has_strategy = ai_strategy_unify_italy
				}
			}
			add = -5
		}
	}
}
