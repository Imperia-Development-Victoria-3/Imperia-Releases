imperia_quick_enact_law = {
	scope = law_type
	saved_scopes = {
		country
	}
	
	effect = {
		scope:country = {
			if = {
				limit = {
					NOT = {
						has_law = prev
					}
				}
				custom_tooltip = {
					text = EXECUTIVE_ORDER_DESC
					activate_law = prev
				}
				add_foreign_policy_cooldown = yes
			}
		}
	}
	
	is_valid = {
		scope:country = {
			NOT = {
				has_law = prev
				custom_tooltip = {
					text = CANNOT_HAVE_EXECUTIVE_ORDER_COOLDOWN
					has_variable = modifier_imperia_foreign_policy_cooldown
				}
			}
			any_law = {
				type = root
				law_is_available = yes
			}
		}
	}
	
	is_shown = {
		scope:country = {
			any_law = {
				type = root
				law_is_available = yes
			}
		}
	}
}
