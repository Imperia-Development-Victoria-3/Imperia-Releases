# For Pretender Israel
coa_def_pretender_israel_flag_trigger = {
	exists = scope:actor
	scope:actor ?= { coa_def_pretender_israel_trigger = yes }
}

coa_def_pretender_israel_trigger = {
	country_has_primary_culture = cu:beta_israel
	has_variable = is_pretender_israel
	NOT = { exists = c:ISR }
	any_country = {
		has_variable = is_pretender_israel
		count = 1
	}
}
