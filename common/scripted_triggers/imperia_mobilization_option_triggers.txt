can_use_mobilization_option_flintlocks = {
	OR = {
		has_technology_researched = gunsmithing
		any_scope_treaty = {
			any_scope_article = {
				has_type = military_assistance
				target_country ?= root
				custom_tooltip = {
					text = MILITARY_ASSISTANCE_SOURCE_MUST_HAVE_RESEARCHED_GUNSMITHING
					source_country ?= {
						has_technology_researched = gunsmithing
					}
				}
			}
		}
	}
}
