imperia_sinosphere.1 = {
	type = country_event
	placement = root
	title = imperia_sinosphere.1.t
	desc = imperia_sinosphere.1.d
	flavor = imperia_sinosphere.1.f
	duration = 3
	event_image = {
		video = "gfx/event_pictures/unspecific_ruler_speaking_to_people.bk2"
	}
	
	on_created_soundeffect = "event:/SFX/UI/Alerts/event_appear"
	on_opened_soundeffect = "event:/SFX/Events/europenorthamerica/rich_and_poor"
	icon = "gfx/interface/icons/event_icons/event_default.dds"
	trigger = {}
	
	option = {
		name = imperia_sinosphere.1.a
	}
	
	option = {
		name = imperia_sinosphere.1.b
		default_option = yes
		
		add_modifier = {
			name = modifier_mixed_script_korea
		}
		add_modifier = {
			name = modifier_writing_system_transition
			months = 60
			is_decaying = yes
		}
		if = {
			limit = {
				AND = {
					exists = c:CHI
					is_subject_of = c:CHI
				}
			}
			create_diplomatic_catalyst = {
				type = catalyst_event_negative
				target = c:CHI
			}
			change_relations = {
				country = c:CHI
				value = -50
			}
		}
		if = {
			limit = {
				AND = {
					exists = c:TPG
					is_subject_of = c:TPG
				}
			}
			create_diplomatic_catalyst = {
				type = catalyst_event_negative
				target = c:TPG
			}
			change_relations = {
				country = c:TPG
				value = -50
			}
		}
	}
	
	option = {
		name = imperia_sinosphere.1.c
		
		trigger = {
			NOR = {
				is_subject_of = c:CHI
				is_subject_of = c:TPG
			}
		}
		
		add_modifier = {
			name = modifier_hangul
		}
		add_modifier = {
			name = modifier_writing_system_transition
			months = 120
			is_decaying = yes
		}
	}
}

		name = imperia_sinosphere.3.b	#The Uighurs take back control

		add_primary_culture = cu:uighur

			culture = cu:han
			value = 0.2
		}
		add_loyalists = {
			culture = cu:uighur
			value = 0.1
		}
		
	}
	
	option = {

		add_primary_culture = cu:uighur

			culture = cu:uighur
			value = 0.1
		}
		add_radicals = {
			culture = cu:han
			value = 0.1
		}
		
	}
}

imperia_sinosphere.4 = {	#Prince Mongkut leaves monkhood?
	type = country_event
	placement = root
	title = imperia_sinosphere.4.t
	desc = imperia_sinosphere.4.d
	flavor = imperia_sinosphere.4.f
	gui_window = event_window_1char_tabloid
	left_icon = scope:king_nanklao_scope
	on_created_soundeffect = "event:/SFX/UI/Alerts/event_appear"
	icon = "gfx/interface/icons/law_icons/monarchy.dds"
	duration = 3
	cooldown = {
		years = stupidly_long_modifier_time
	}
	trigger = {
		c:SIA ?= this
		c:SIA = {
			has_law = law_type:law_monarchy
			ruler ?= {
				has_template = SIA_nangklao_chakri
			}
		}
		game_date >= 1851.4.2
	}
	immediate = {
		ruler = {
			save_scope_as = king_nanklao_scope
		}
	}
	option = {
		name = imperia_sinosphere.4.a	#Mongkut defrocks and becomes King
		default_option = yes
		custom_tooltip = {
			text = mongkut_enthroned
		}
		scope:king_nanklao_scope = {
			kill_character = yes
		}
		create_character = {
			template = imperia_sia_king_mongkut_template
			ruler = yes
		}
		hidden_effect = {
			trigger_event = {
				id = imperia_sinosphere.6
				days = 950
			}
		}
		change_relations = {
			country = c:GBR
			value = 20
		}
		create_diplomatic_catalyst = {
			type = catalyst_relations_level_increased
			target = c:GBR
		}
	}
	option = {
		name = imperia_sinosphere.4.b	#Pinklao takes the throne
		scope:king_nanklao_scope = {
			kill_character = yes
		}
		custom_tooltip = {
			text = pinklao_enthroned
		}
	}
}

imperia_sinosphere.5 = {	#Mindon's Coup
	type = country_event
	placement = root
	title = imperia_sinosphere.5.t
	desc = imperia_sinosphere.5.d
	flavor = imperia_sinosphere.5.f
	gui_window = event_window_1char_tabloid
	left_icon = scope:burma_king
	on_created_soundeffect = "event:/SFX/UI/Alerts/event_appear"
	icon = "gfx/interface/icons/law_icons/monarchy.dds"
	duration = 3
	cooldown = {
		years = stupidly_long_modifier_time
	}
	trigger = {
		c:BUR ?= this
		has_law = law_type:law_monarchy
		NOT = {
			any_state = {
				state_region = s:STATE_PEGU
				owner = c:BUR
			}
		}
		game_date >= 1853.1.1
	}
	immediate = {
		ruler = {
			save_scope_as = burma_king
		}
	}
	option = {
		name = imperia_sinosphere.5.a
		default_option = yes
		scope:burma_king = {
			kill_character = yes
		}
		create_character = {
			template = imperia_bur_mindon_min_template
			ruler = yes
		}
		create_character = {
			template = imperia_bur_kananung_mintha_template
			heir = yes
		}
		add_radicals = {
			value = 0.1
		}
	}
	option = {
		name = imperia_sinosphere.5.b
		add_radicals = {
			value = 0.1
		}
	}
}
